题目分析

DP 方程很显然 但是发现可能互相更新 也就是说这个DP是有后效性的 怎么办?类似spfa的方法,一个点如果被更新 将它所有的fa再次压入队列进行更新

#include <cstdio>
#include <cstring>
#include <cmath>
#include <queue>
#include <vector>
#include <algorithm>
using namespace std;
int n;
vector<int> v[200010];
long long a[200010],b[200010];
int tot,head[200010],to[4000010],net[4000010];
void add(int x,int y) 
{ net[++tot]=head[x],head[x]=tot,to[tot]=y; }
bool vis[200010];
void spfa()
{
    queue<int>q;
    for(int i=1;i<=n;i++) vis[i]=1,q.push(i);
    while(q.size())
    {   
        int x=q.front();
        q.pop(),vis[x]=0;
        long long tmp=a[x];
        for(int i=head[x];i;i=net[i]) tmp+=b[to[i]];
        if(tmp>=b[x])continue;
        b[x]=tmp;
        for(int i=0;i<v[x].size();i++)
            if(!vis[v[x][i]])
                vis[v[x][i]]=1,q.push(v[x][i]);
    }
}
int main()
{
    scanf("%d",&n);
    for(int x,tmp,i=1;i<=n;i++) 
    {
        scanf("%lld%lld%d",&a[i],&b[i],&tmp);
        while(tmp--) scanf("%d",&x),add(i,x),v[x].push_back(i);
    }
    spfa();
    printf("%lld",b[1]);
    return 0;
}

初探区间DP f[i][j]代表将区间ij涂成对应颜色最少次数 假设枚举到区间lr 如果l的颜色和r的颜色相同 那么相当于在最开始在l~r的位置涂了一次 如果l r 挨在一起 f[l][r]=1 否则f[l][r]=f[l+1][r-1]+1,f[l][r-1],f[l+1][r]; 如果l的颜色和r的颜色不同 那么我们枚举中点分割区间 f[l][r]=f[l][i]+f[i+1][r]

题目分析

这题省选竟然爆零 真TM丧心病狂 三分截止时间 O(N)check 并且这题可以做到O(N); 注意特判c>10^16的情况

#include <cstdio>
#include <cstring>
#include <set>
#include <map>
#include <vector>
#include <cmath>
#include <queue>
#include <algorithm>
using namespace std;
int n,m;
int a[100010],b[100010];
long long A,B,C,ans;
long long check(int mid)
{
    long long sum=0,ltim=0,rtim=0;
    for(int i=1;i<=m;i++)
        if(b[i]<=mid) ltim+=(mid-b[i]);  
        else rtim+=(b[i]-mid);
    if(ltim>=rtim) sum=sum+(long long) min(A,B)*rtim;
    else
    {    
        sum=sum+(long long) B*(rtim-ltim);
        sum=sum+(long long) min(A,B)*ltim;
    } 
    for(int i=1;i<=n;i++) 
        if(mid>a[i]) sum=sum+(long long) C*(mid-a[i]);
    return sum;
}
int main()
{    
    scanf("%lld%lld%lld",&A,&B,&C);
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    for(int i=1;i<=m;i++) scanf("%d",&b[i]);
    sort(a+1,a+n+1),sort(b+1,b+m+1);
    ans=23333333333333333ll;
    if(C>2147483647) printf("%lld",check(a[1]));
    else
    {
        int l=0,r=100010;
        while(1)
        {
            if(r-l<=10)
            {
                for(int i=l;i<=r;i++) ans=min(check(i),ans);
                break;    
            }
            int ll=(l+(r-l)/3),rr=(r-(r-l)/3);
            if(check(ll)<check(rr)) r=rr;
            else l=ll;
        }
        printf("%lld",ans);
    }
    return 0;

Description

两只青蛙在网上相识了，它们聊得很开心，于是觉得很有必要见一面。它们很高兴地发现它们住在同一条纬度线上，于是它们约定各自朝西跳，直到碰面为止。可是它们出发之前忘记了一件很重要的事情，既没有问清楚对方的特征，也没有约定见面的具体位置。不过青蛙们都是很乐观的，它们觉得只要一直朝着某个方向跳下去，总能碰到对方的。但是除非这两只青蛙在同一时间跳到同一点上，不然是永远都不可能碰面的。为了帮助这两只乐观的青蛙，你被要求写一个程序来判断这两只青蛙是否能够碰面，会在什么时候碰面。 我们把这两只青蛙分别叫做青蛙A和青蛙B，并且规定纬度线上东经0度处为原点，由东往西为正方向，单位长度1米，这样我们就得到了一条首尾相接的数轴。设青蛙A的出发点坐标是x，青蛙B的出发点坐标是y。青蛙A一次能跳m米，青蛙B一次能跳n米，两只青蛙跳一次所花费的时间相同。纬度线总长L米。现在要你求出它们跳了几次以后才会碰面。

Input

输入只包括一行5个整数x，y，m，n，L，其中x≠y < 2000000000，0 < m、n < 2000000000，0 < L < 2100000000。

Output

输出碰面所需要的跳跃次数，如果永远不可能碰面则输出一行”Impossible”

Sample Input

1 2 3 4 5

Sample Output

4

题目分析

令k为跳跃最小次数 $$ x+km=y+kn(\\%L)$$ $$ k(m-n)=(y-x)(\\%L)$$ 然后就是扩展gcd求解同余方程最小正整数解

#include<cstdio>
#include<algorithm>
using namespace std;
long long n,m,ans,tmp;
long long x,y,a,b,c,d;
long long exgcd(long long a,long long b)
{
    if(b==0)
    {
        x=1;
        y=0;
        return a;
    }
    d=exgcd(b,a%b);
    tmp=x;
    x=y;
    y=tmp-(a/b)*y;
    return d;
}
int main()
{
    scanf("%lld%lld%lld%lld%lld",&x,&y,&m,&n,&b);
    a=((m-n)%b+b)%b;
    c=((y-x)%b+b)%b;
    d=exgcd(a,b);
    if(c%d!=0)
    {
        printf("Impossible");
        return 0;
    }
    a/=d;b/=d;c/=d;
    x=(x%b+b)%b;
    x=(x*c)%b;
    printf("%lld\n",x);
    return 0;
}

7 4
5 7 2 4 3 1 6

Sample Output

4

HINT

第三个样例解释：{4}, {7,2,4}, {5,7,2,4,3}和{5,7,2,4,3,1,6} N<=100000

题目分析

找到b所在位置 考虑一个数比x大，能对答案造成什么影响，一个数比x小，能对答案造成什么影响，然后枚举子序列长度统计答案即可

#include <iostream>
#include <cstdio>
using namespace std;
int b,n,a[400001],pos,x,sum[200001],ans;
int l[400001],r[400001];
int main()
{
    scanf("%d%d",&n,&b);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&x);
        if(x==b) pos=i,a[i]=0;
        else if(x>b)a[i]=1;
        else a[i]=-1;
    }
    l[n]=1;r[n]=1;
    for(int i=pos-1;i>=1;i--)
        sum[i]=sum[i+1]+a[i],l[sum[i]+n]++;
    for(int i=pos+1;i<=n;i++)
        sum[i]=sum[i-1]+a[i],r[sum[i]+n]++;
    for(int i=-n+n;i<=n-1+n;i++)
        ans+=l[i]*r[n-i+n];
    printf("%d",ans);
}

{

        int x=q.front();q.pop();

一个整数，为答案模1000000007后的值。

题目分析

$$ \\sum\\limits\_{i=1}^{n}\\sum\\limits\_{j=1}^{m} lcm(i,j)^{gcd(i,j)}$$ $$ =\\sum\\limits\_{d=1}^{n} \\sum\\limits\_{i=1}^{n}\\sum\\limits\_{j=1}^{m} (\\frac{ij}{d})^d[gcd(i,j)=d]$$ $$ =\\sum\\limits\_{d=1}^{n}\\sum\\limits\_{i=1}^{\\lfloor \\frac{n}{d} \\rfloor }\\sum\\limits\_{j=1}^{\\lfloor \\frac{m}{d} \\rfloor}(\\frac{id\*jd}{d})^d[gcd(i,j)=1]$$ $$ =\\sum\\limits\_{d=1}^{n}\\sum\\limits\_{i=1}^{\\lfloor \\frac{n}{d} \\rfloor}\\sum\\limits\_{j=1}^{\\lfloor \\frac{m}{d} \\rfloor}(ijd)^d\\sum\\limits\_{k|gcd(i,j)}^{}\\mu(k)$$ $$ =\\sum\\limits\_{d=1}^{n}d^d\\sum\\limits\_{k=1}^{\\lfloor \\frac{n}{d} \\rfloor}k^{2d}\\mu(k)\(\sum\\limits\_{i=1}^{\\lfloor \\frac{n}{dk} \\rfloor}i^d )(\\sum\\limits\_{j=1}^{\\lfloor \\frac{m}{d} \\rfloor}j^d)$$ 然后暴力更新即可 时间复杂度 n ln n
