[BZOJ3262] 陌上花开

Posted on 2017-09-14 Edited on 2020-07-05

题目描述

Description

有n朵花，每朵花有三个属性：花形(s)、颜色(c)、气味(m)，又三个整数表示。现要对每朵花评级，一朵花的级别是它拥有的美丽能超过的花的数量。定义一朵花A比另一朵花B要美丽，当且仅当Sa>=Sb,Ca>=Cb,Ma>=Mb。显然，两朵花可能有同样的属性。需要统计出评出每个等级的花的数量。

Input

第一行为N,K (1 <= N <= 100,000, 1 <= K <= 200,000 ), 分别表示花的数量和最大属性值。 以下N行，每行三个整数si, ci, mi (1 <= si, ci, mi <= K)，表示第i朵花的属性

Output

包含N行，分别表示评级为0…N-1的每级花的数量。

Sample Input

10 3
3 3 3
2 3 3 
2 3 1 
3 1 1 
3 1 2 
1 3 1 
1 1 2 
1 2 2 
1 3 2 
1 2 1

Sample Output

3
1
3
0
1
0
1
0
0
1

HINT

1 <= N <= 100,000, 1 <= K <= 200,000

题目分析

CDQ维护三维偏序裸题 就是去重太恶心了 QAQ

#include <cstdio>
#include <cstring>
#include <set>
#include <map>
#include <vector>
#include <cmath>
#include <queue>
#include <algorithm>
using namespace std;
int n,k;
struct your
{
    int s,c,m;
    int ans,num;
}a[100100];
bool cmp(your j,your k)
{
   return (j.s==k.s)?((j.c==k.c)?j.m<k.m:j.c<k.c):j.s<k.s;
}
bool cmp2(your j,your k)
{
    //if(j.c==k.c) return j.m<k.m;
    return j.c<k.c;
}
int f[201000];
void update(int x,int c)
{
    for(;x<=k;x+=x&-x) f[x]+=c;
}
int ask(int x)
{
    int sum=0;
    for(;x;x-=x&-x) sum+=f[x];
    return sum;
}
void clean(int x)
{
    for(;x<=k;x+=x&-x) f[x]=0;
}
void cdq(int l,int r)
{
    if(l==r) return ;
    int mid=(l+r)>>1;
    cdq(l,mid),cdq(mid+1,r);
    sort(a+l,a+mid+1,cmp2),sort(a+mid+1,a+r+1,cmp2);
    int tmp=l,nmp=mid+1;
    for(;nmp<=r;nmp++)
    {
        while(tmp<=mid&&a[tmp].c<=a[nmp].c)
            update(a[tmp].m,a[tmp].num),tmp++;
        a[nmp].ans+=ask(a[nmp].m);
    }
    for(int i=l;i<=mid;i++) clean(a[i].m);
}
int kkk;
int temp[1000010];
int main() 
{
    scanf("%d%d",&n,&k),kkk=n;
    for(int i=1;i<=n;i++) scanf("%d%d%d",&a[i].s,&a[i].c,&a[i].m);
    int cnt=0,now=1;
    sort(a+1,a+n+1,cmp);
    a[++cnt]=a[1];
    for(int i=2;i<=n;i++)
    {
        if(a[i].s!=a[i-1].s||a[i].c!=a[i-1].c||a[i].m!=a[i-1].m)
            a[++cnt]=a[i],a[cnt-1].num=now,now=1;
        else now++;
    }
    a[cnt].num=now;
    cdq(1,cnt);
    for(int i=1;i<=cnt;++i)
        temp[a[i].ans+a[i].num-1]+=a[i].num;
    for(int i=0;i<n;i++)printf("%d\n",temp[i]);
    return 0;
}
